CREATE TABLE Users
(
	UID int NOT NULL AUTO_INCREMENT,
	Username varchar (30),
	Password char (60),
	Email varchar (254),
	PRIMARY KEY (UID)
);

CREATE TABLE Team
(
	TeamID int NOT NULL AUTO_INCREMENT,
	Name varchar (30),
	HeadCoach int,
	CONSTRAINT HeadCoach_fk FOREIGN KEY (HeadCoach) REFERENCES Users (UID),
	PRIMARY KEY (TeamID)
);
